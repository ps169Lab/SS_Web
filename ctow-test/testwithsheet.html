<!DOCTYPE html>

<html lang="en">

<head><meta charset="UTF-8"><title>Google Sheet JSON to HTML</title></head>

<body>

<h2>Google Sheet Data</h2>

<table id="dataTable" border="1"></table>

<script>

const sheetId = 'your_spreadsheet_id_here';

const sheetNumber = 1;

const url = `https://docs.google.com/spreadsheets/d/1A31A5GORE1V75T7IuoD8MG6IZh52sHr0FZbXKbM4v8c/edit?usp=sharing`;


fetch(url)

  .then(response => response.json())

  .then(data => {

    const entries = data.feed.entry;

    const table = document.getElementById('dataTable');

    if(entries.length > 0){

      // Create table header from keys

      const headers = Object.keys(entries[0]).filter(k => k.startsWith("gsx$"));

      const headerRow = document.createElement('tr');

      headers.forEach(h => {

        const th = document.createElement('th');

        th.textContent = h.replace('gsx$', '');

        headerRow.appendChild(th);

      });

      table.appendChild(headerRow);


      // Create table rows

      entries.forEach(entry => {

        const tr = document.createElement('tr');

        headers.forEach(h => {

          const td = document.createElement('td');

          td.textContent = entry[h].$t;

          tr.appendChild(td);

        });

        table.appendChild(tr);

      });

    } else {

      table.textContent = 'No data found';

    }

  })

  .catch(e => {

    console.error(e);

    document.getElementById('dataTable').textContent = 'Failed to load data';

  });

</script>

</body>

</html>
